if(BHARAT_HOST_TEST OR BHARAT_BUILD_UNIT_TESTS)
    set(BHARAT_SYSCALL_SOURCES host_syscall_stub.c)
else()
    set(BHARAT_SYSCALL_SOURCES "")
    if(BHARAT_ARCH_HAS_SDK_SYSCALL_BACKEND)
        if(BHARAT_ARCH_FAMILY MATCHES "RISCV" AND BHARAT_ARCH_BITS EQUAL 64)
            list(APPEND BHARAT_SYSCALL_SOURCES ${CMAKE_SOURCE_DIR}/experience/user/sdk/lib/syscall/syscall_riscv64.S)
            set(BHARAT_HAS_ARCH_SYSCALL_BACKEND 1)
        elseif(BHARAT_ARCH_FAMILY MATCHES "RISCV" AND BHARAT_ARCH_BITS EQUAL 32)
            list(APPEND BHARAT_SYSCALL_SOURCES ${CMAKE_SOURCE_DIR}/experience/user/sdk/lib/syscall/syscall_riscv32.S)
            set(BHARAT_HAS_ARCH_SYSCALL_BACKEND 1)
        elseif(BHARAT_ARCH_FAMILY MATCHES "ARM" AND BHARAT_ARCH_BITS EQUAL 64)
            list(APPEND BHARAT_SYSCALL_SOURCES ${CMAKE_SOURCE_DIR}/experience/user/sdk/lib/syscall/syscall_arm64.S)
            set(BHARAT_HAS_ARCH_SYSCALL_BACKEND 1)
        elseif(BHARAT_ARCH_FAMILY MATCHES "ARM" AND BHARAT_ARCH_BITS EQUAL 32)
            list(APPEND BHARAT_SYSCALL_SOURCES ${CMAKE_SOURCE_DIR}/experience/user/sdk/lib/syscall/syscall_arm32.S)
            set(BHARAT_HAS_ARCH_SYSCALL_BACKEND 1)
        elseif(BHARAT_ARCH_FAMILY MATCHES "X86" AND BHARAT_ARCH_BITS EQUAL 64)
            list(APPEND BHARAT_SYSCALL_SOURCES ${CMAKE_SOURCE_DIR}/experience/user/sdk/lib/syscall/syscall_x86_64.S)
            set(BHARAT_HAS_ARCH_SYSCALL_BACKEND 1)
        endif()
    endif()

    if(BHARAT_ENABLE_USERSPACE AND NOT BHARAT_HAS_ARCH_SYSCALL_BACKEND)
        message(FATAL_ERROR "Userspace enabled but no syscall backend found for ${BHARAT_ARCH_FAMILY}-${BHARAT_ARCH_BITS}")
    endif()
endif()

if(BHARAT_SYSCALL_SOURCES)
    add_library(bharat_syscall STATIC ${BHARAT_SYSCALL_SOURCES})

    if(BHARAT_HAS_ARCH_SYSCALL_BACKEND)
        target_compile_definitions(bharat_syscall PRIVATE BHARAT_HAS_ARCH_SYSCALL_BACKEND=1)
    endif()
    target_include_directories(bharat_syscall PUBLIC ${CMAKE_SOURCE_DIR}/interface/include)
    target_link_libraries(bharat_syscall PUBLIC bharat_freestanding)
endif()
