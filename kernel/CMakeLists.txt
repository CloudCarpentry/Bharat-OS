    set(BHARAT_MANIFEST_JSON "${CMAKE_CURRENT_SOURCE_DIR}/../release/manifest/os-release.json")
    set(BHARAT_GEN_SCRIPT "${CMAKE_CURRENT_SOURCE_DIR}/../tools/gen_manifest.py")
    set(BHARAT_GEN_DIR "${CMAKE_BINARY_DIR}/generated")

    if(EXISTS "${BHARAT_MANIFEST_JSON}" AND EXISTS "${BHARAT_GEN_SCRIPT}")
        find_package(Python3 QUIET COMPONENTS Interpreter)
        if(Python3_Interpreter_FOUND)
            file(MAKE_DIRECTORY "${BHARAT_GEN_DIR}")
            execute_process(
                COMMAND "${Python3_EXECUTABLE}" "${BHARAT_GEN_SCRIPT}"
                        "${BHARAT_MANIFEST_JSON}" "${BHARAT_GEN_DIR}"
                WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/.."
                RESULT_VARIABLE BHARAT_GEN_RESULT
            )
            if(BHARAT_GEN_RESULT EQUAL 0 AND EXISTS "${BHARAT_GEN_DIR}/BharatVersion.cmake")
                include("${BHARAT_GEN_DIR}/BharatVersion.cmake")
            endif()
        endif()
    endif()
endif()

    ${KERNEL_SRCS}
    $<TARGET_OBJECTS:bharat_mm_iommu>
    $<TARGET_OBJECTS:bharat_mm_pmm>
    $<TARGET_OBJECTS:k_core>
    $<TARGET_OBJECTS:k_sched>
    $<TARGET_OBJECTS:k_ipc>
    $<TARGET_OBJECTS:k_cap>
    $<TARGET_OBJECTS:k_trap>
    $<TARGET_OBJECTS:k_debug>
    $<TARGET_OBJECTS:k_init>
    $<TARGET_OBJECTS:k_fs>
)

if(BHARAT_ENABLE_KERNEL_SELFTESTS)
    target_sources(kernel.elf PRIVATE $<TARGET_OBJECTS:k_test_support>)
endif()

if(NOT BHARAT_PROFILE_MPU_ONLY)
    target_sources(kernel.elf PRIVATE
        $<TARGET_OBJECTS:bharat_mm_aspace>
        $<TARGET_OBJECTS:bharat_mm_objects>
        $<TARGET_OBJECTS:bharat_mm_fault>
        $<TARGET_OBJECTS:bharat_mm_pt>
        $<TARGET_OBJECTS:bharat_mm_tlb>
    )
endif()
target_sources(kernel.elf PRIVATE
    $<TARGET_OBJECTS:bharat_mm_vmm>
)
# Architecture-specific code model and PIE flags are set below
target_link_libraries(kernel.elf subsys_manager bharat_libmsg bharat_kernel_buildopts bharatos_drivers bharat_kernel_includes board_fabric)
if(BHARAT_ENABLE_FBUI)
    target_link_libraries(kernel.elf fbui)
endif()
set_target_properties(kernel.elf PROPERTIES
    OUTPUT_NAME "kernel.elf"
    SUFFIX ""
)
# Extra warnings for C sources only
target_compile_options(kernel.elf PRIVATE
    $<$<COMPILE_LANGUAGE:C>:-ffreestanding>
    $<$<COMPILE_LANGUAGE:C>:-fno-builtin>
    $<$<COMPILE_LANGUAGE:C>:-Wall>
    $<$<COMPILE_LANGUAGE:C>:-Wextra>
    $<$<COMPILE_LANGUAGE:C>:-Werror=implicit-function-declaration>
)
option(BHARAT_KERNEL_HARDENING "Enable kernel hardening compile/link options" ON)
option(BHARAT_KERNEL_ASLR "Build kernel as PIE for KASLR-friendly relocation experiments" OFF)
if(BHARAT_KERNEL_HARDENING)
    target_compile_options(kernel.elf PRIVATE
        $<$<COMPILE_LANGUAGE:C>:-fstack-protector-strong>
        $<$<COMPILE_LANGUAGE:C>:-fno-strict-aliasing>
        $<$<COMPILE_LANGUAGE:C>:-fno-delete-null-pointer-checks>
    )
    target_link_options(kernel.elf PRIVATE
        "-z"
        "noexecstack"
    )
else()
    target_compile_options(kernel.elf PRIVATE
        $<$<COMPILE_LANGUAGE:C>:-fno-stack-protector>
    )
endif()
if(BHARAT_KERNEL_ASLR)
    target_compile_options(kernel.elf PRIVATE
        $<$<COMPILE_LANGUAGE:C>:-fpie>
    )
    target_link_options(kernel.elf PRIVATE
        "LINKER:--pie"
    )
else()
    target_compile_options(kernel.elf PRIVATE
        $<$<COMPILE_LANGUAGE:C>:-fno-pie>
    )
    target_link_options(kernel.elf PRIVATE
        "LINKER:--no-pie"
    )
endif()
if(ARCH STREQUAL "x86_64")
    target_compile_options(kernel.elf PRIVATE
        $<$<COMPILE_LANGUAGE:C>:-m64>
        $<$<COMPILE_LANGUAGE:C>:-mno-red-zone>
    )
    target_link_options(kernel.elf PRIVATE
        "LINKER:-m,elf_x86_64"
    )
elseif(ARCH MATCHES "riscv.*64")
    target_compile_options(kernel.elf PRIVATE
        $<$<OR:$<COMPILE_LANGUAGE:C>,$<COMPILE_LANGUAGE:ASM>>:-mcmodel=medany>
    )
    target_link_options(kernel.elf PRIVATE
        "LINKER:-m,elf64lriscv"
    )
elseif(ARCH MATCHES "arm.*64" OR ARCH MATCHES "aarch64")
    target_compile_options(kernel.elf PRIVATE
        $<$<OR:$<COMPILE_LANGUAGE:C>,$<COMPILE_LANGUAGE:ASM>>:-mcmodel=small>
    )
    target_link_options(kernel.elf PRIVATE
        "LINKER:-m,aarch64elf"
    )
elseif(ARCH STREQUAL "riscv32")
    target_compile_options(kernel.elf PRIVATE
        $<$<COMPILE_LANGUAGE:C>:-mcmodel=medany>
        $<$<COMPILE_LANGUAGE:ASM>:-march=rv32g>
        $<$<COMPILE_LANGUAGE:C>:-march=rv32g>
    )
    target_link_options(kernel.elf PRIVATE
        "LINKER:-m,elf32lriscv"
    )
endif()
# ── Linker options
# Use LINKER: so CMake can translate options correctly when linking via the
# compiler driver across clang/gcc and host/cross presets.
target_link_options(kernel.elf PRIVATE
    "-nostdlib"
    $<$<AND:$<C_COMPILER_ID:Clang>,$<STREQUAL:${ARCH},arm32>>:-fuse-ld=bfd>
    "-no-pie" "-fno-pie"
    ""
    "LINKER:--fatal-warnings"
    "LINKER:-T,${LINKER_SCRIPT}"
    "LINKER:--build-id=none"
)

# ── Component Registration ──────────────────────────────────────────────────
bharat_register_component(
    TARGET kernel.elf
    KIND "kernel"
    VERSION "${BHARAT_KERNEL_VERSION}"
    IFACE_VERSION "${BHARAT_KERNEL_ABI}"
    CHANNEL "${BHARAT_OS_CHANNEL}"
)
if(ARCH STREQUAL "riscv64" AND BHARAT_RISCV_BUILD_PAYLOAD_BIN)
    if(NOT CMAKE_OBJCOPY)
        find_program(CMAKE_OBJCOPY NAMES llvm-objcopy riscv64-unknown-elf-objcopy)
    endif()
    if(NOT CMAKE_OBJCOPY)
        message(WARNING "objcopy not found; payload.bin target disabled")
    else()
        add_custom_command(
            OUTPUT "${CMAKE_CURRENT_BINARY_DIR}/payload.bin"
            COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:kernel.elf> "${CMAKE_CURRENT_BINARY_DIR}/payload.bin"
            DEPENDS kernel.elf
            COMMENT "Generating OpenSBI payload.bin from kernel.elf"
            VERBATIM
        )
        add_custom_target(kernel.payload.bin ALL DEPENDS "${CMAKE_CURRENT_BINARY_DIR}/payload.bin")
    endif()
endif()
